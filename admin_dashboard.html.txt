    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, onSnapshot, collection, setLogLevel, updateDoc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Initialize Lucide icons
        lucide.createIcons();
        setLogLevel('debug');

        // =========================================================================================
        // IMPORTANT: FIREBASE CONFIGURATION CHECK
        // =========================================================================================
        const YOUR_EXTERNAL_FIREBASE_CONFIG = {
            apiKey: "YOUR_API_KEY_HERE",           
            authDomain: "YOUR_AUTH_DOMAIN_HERE",   
            projectId: "YOUR_PROJECT_ID_HERE",     
            storageBucket: "YOUR_STORAGE_BUCKET_HERE",
            messagingSenderId: "YOUR_MESSAGING_SENDER_ID_HERE",
            appId: "YOUR_APP_ID_HERE"
        };
        // =========================================================================================

        // --- 1. CONFIGURATION HANDLING ---
        const isCanvasMode = typeof __firebase_config !== 'undefined' && __firebase_config.length > 0;
        
        let config;
        let authType = 'ANONYMOUS'; 
        let globalAppId = 'lodge-membership-app';

        if (isCanvasMode) {
            config = JSON.parse(__firebase_config);
            globalAppId = typeof __app_id !== 'undefined' ? __app_id : 'default-canvas-app-id';
            authType = 'CUSTOM_TOKEN';
        } else {
            config = YOUR_EXTERNAL_FIREBASE_CONFIG;
            globalAppId = config.projectId; 
            
            if (config.apiKey === "YOUR_API_KEY_HERE" || !config.projectId || config.projectId === "YOUR_PROJECT_ID_HERE") {
                 console.error("EXTERNAL RUN ERROR: Please update YOUR_EXTERNAL_FIREBASE_CONFIG with valid keys.");
                 config = null; 
            }
        }

        let db, auth;
        let userId = null;
        let applications = []; // Storage for all applications
        let currentEditingDocId = null; // Track which document is being edited
        
        // UI Elements
        const statusMessage = document.getElementById('statusMessage');
        const errorDisplay = document.getElementById('errorDisplay');
        const errorText = document.getElementById('errorText');
        const adminDashboard = document.getElementById('adminDashboard');
        const applicationsTableBody = document.getElementById('applicationsTableBody');
        const emptyState = document.getElementById('emptyState');
        const adminUserIdEl = document.getElementById('admin-user-id');
        
        // Modal Elements
        const modal = document.getElementById('applicationModal');
        const applicantNameModal = document.getElementById('applicantNameModal');
        const applicantDataDisplay = document.getElementById('applicantDataDisplay');
        const adminUpdateForm = document.getElementById('adminUpdateForm');
        const updateMessage = document.getElementById('updateMessage');
        const updateButton = document.getElementById('updateButton');


        /**
         * Initializes Firebase services and authenticates the user.
         */
        async function initFirebase() {
            try {
                if (!config) {
                    throw new Error("Initialization blocked: Firebase configuration is missing or invalid.");
                }

                // 2. Initialize services
                const app = initializeApp(config);
                auth = getAuth(app);
                db = getFirestore(app);

                let userCredential;

                // 3. Authenticate
                if (authType === 'CUSTOM_TOKEN') {
                    const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;
                    if (!initialAuthToken) {
                        userCredential = await signInAnonymously(auth); 
                    } else {
                        userCredential = await signInWithCustomToken(auth, initialAuthToken);
                    }
                } else {
                    userCredential = await signInAnonymously(auth);
                }

                if (userCredential && userCredential.user && userCredential.user.uid) {
                    userId = userCredential.user.uid;
                    adminUserIdEl.textContent = `Admin ID: ${userId}`;
                    console.log(`Admin Portal Authentication SUCCESS (${authType}). User ID: ${userId}`);
                    
                    // Success: Setup data listener
                    statusMessage.classList.add('hidden');
                    adminDashboard.classList.remove('hidden');
                    listenForApplications();
                    
                } else {
                    throw new Error("Authentication failed or returned no user.");
                }

            } catch (error) {
                console.error("Firebase Initialization Error:", error);
                
                let friendlyError = error.message.includes('apiKey') 
                    ? "FATAL ERROR: Configuration is missing or invalid. Please check the code."
                    : error.message;

                statusMessage.classList.add('hidden');
                errorDisplay.classList.remove('hidden');
                errorText.textContent = friendlyError;
            }
        }
        
        /**
         * Listens to the 'applications' collection for real-time updates.
         */
        function listenForApplications() {
            // Path: /artifacts/{appId}/public/data/applications
            const collectionPath = `/artifacts/${globalAppId}/public/data/applications`;
            const applicationsRef = collection(db, collectionPath);
            const q = query(applicationsRef); // You can add orderBy here if needed, but it may require composite indexes

            onSnapshot(q, (snapshot) => {
                applications = [];
                snapshot.forEach((doc) => {
                    applications.push({ id: doc.id, ...doc.data() });
                });
                renderApplicationsTable();
            }, (error) => {
                console.error("Error listening to applications: ", error);
                errorDisplay.classList.remove('hidden');
                errorText.textContent = `Database Read Error: ${error.message}`;
            });
        }

        /**
         * Renders the applications list into the table.
         */
        function renderApplicationsTable() {
            applicationsTableBody.innerHTML = '';
            
            if (applications.length === 0) {
                emptyState.classList.remove('hidden');
                return;
            } else {
                emptyState.classList.add('hidden');
            }

            applications.forEach(app => {
                const statusColor = getStatusColor(app.status);
                const submissionDate = app.submissionDate ? new Date(app.submissionDate.seconds * 1000).toLocaleDateString() : 'N/A';
                
                const row = document.createElement('tr');
                row.className = 'table-row-hover';
                row.onclick = () => openModal(app.id);

                row.innerHTML = `
                    <td class="px-6 py-4 whitespace-nowrap text-sm font-medium text-gray-900">
                        ${app.printName || 'No Name Provided'}
                    </td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 hidden sm:table-cell">
                        ${app.lodgeNumber || 'N/A'}
                    </td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500 hidden md:table-cell">
                        ${submissionDate}
                    </td>
                    <td class="px-6 py-4 whitespace-nowrap">
                        <span class="px-2 inline-flex text-xs leading-5 font-semibold rounded-full ${statusColor}">
                            ${app.status}
                        </span>
                    </td>
                `;
                applicationsTableBody.appendChild(row);
            });

            // Re-initialize icons in case any were missed
            lucide.createIcons(); 
        }

        /**
         * Helper to get Tailwind classes based on status.
         */
        function getStatusColor(status) {
            switch (status) {
                case 'Pending Review':
                    return 'bg-yellow-100 text-yellow-800';
                case 'Investigation Committee':
                    return 'bg-blue-100 text-blue-800';
                case 'Approved':
                case 'Completed':
                    return 'bg-green-100 text-green-800';
                case 'Rejected':
                    return 'bg-red-100 text-red-800';
                default:
                    return 'bg-gray-100 text-gray-800';
            }
        }

        /**
         * Opens the detail modal for a specific application.
         */
        window.openModal = function(docId) {
            currentEditingDocId = docId;
            const app = applications.find(a => a.id === docId);

            if (!app) {
                console.error("Application not found:", docId);
                return;
            }

            // 1. Populate Read-Only Details
            applicantNameModal.textContent = app.printName || 'Unknown Applicant';
            applicantDataDisplay.innerHTML = `
                <div class="grid grid-cols-2 gap-4">
                    <p><strong>Submitted By ID:</strong> ${app.applicantId}</p>
                    <p><strong>Lodge Name:</strong> ${app.lodgeName}</p>
                    <p><strong>Lodge No:</strong> ${app.lodgeNumber}</p>
                    <p><strong>Applicant Phone:</strong> ${app.applicantPhone}</p>
                    <p><strong>Wife Name:</strong> ${app.wifeName || 'N/A'}</p>
                    <p><strong>Occupation:</strong> ${app.occupation || 'N/A'}</p>
                    <p><strong>Health Status:</strong> ${app.healthStatus || 'N/A'}</p>
                    <p><strong>Convicted Crime:</strong> ${app.convictedCrime || 'N/A'}</p>
                    <p><strong>Reference 1:</strong> ${app.reference1 || 'N/A'}</p>
                </div>
                <div class="pt-4 border-t mt-4">
                    <p class="font-semibold text-gray-700">Full Address: ${app.streetAddress}, ${app.applicantCity}, ${app.applicantState} ${app.applicantZip}</p>
                </div>
            `;
            
            // 2. Populate Admin Form (and clear messages)
            adminUpdateForm.reset();
            updateMessage.textContent = '';
            
            // Set values from existing document data
            document.getElementById('currentStatus').value = app.status || 'Pending Review';
            document.getElementById('committee1').value = app.committee1 || '';
            document.getElementById('committee2').value = app.committee2 || '';
            document.getElementById('committee3').value = app.committee3 || '';
            document.getElementById('ballotDate').value = app.ballotDate || '';
            document.getElementById('adminApproved').value = app.adminApproved || '';

            // 3. Show Modal
            modal.classList.remove('hidden');
            modal.style.display = 'flex';
        }

        /**
         * Closes the detail modal.
         */
        window.closeModal = function() {
            modal.style.display = 'none';
            modal.classList.add('hidden');
            currentEditingDocId = null;
        }

        /**
         * Handles the administrative update submission.
         */
        adminUpdateForm.addEventListener('submit', async function(event) {
            event.preventDefault();
            
            if (!currentEditingDocId) {
                updateMessage.textContent = 'Error: No document selected for update.';
                updateMessage.className = 'text-red-500 mt-2';
                return;
            }

            updateButton.disabled = true;
            updateButton.textContent = 'Saving...';
            updateMessage.textContent = '';

            try {
                const formData = new FormData(adminUpdateForm);
                const updates = {};
                for (let [key, value] of formData.entries()) {
                    updates[key] = value;
                }
                
                // Rename 'currentStatus' field to 'status' for consistency
                if (updates.currentStatus) {
                    updates.status = updates.currentStatus;
                    delete updates.currentStatus;
                }

                // Reference the document
                const docRef = doc(db, `/artifacts/${globalAppId}/public/data/applications`, currentEditingDocId);
                
                // Update the document
                await updateDoc(docRef, updates);

                updateMessage.textContent = 'Record successfully updated!';
                updateMessage.className = 'text-green-600 font-bold mt-2';

            } catch (e) {
                console.error("Error updating document: ", e);
                updateMessage.textContent = `Update failed: ${e.message}`;
                updateMessage.className = 'text-red-500 mt-2';
            } finally {
                updateButton.textContent = 'Update Administrative Record';
                updateButton.disabled = false;
                // Keep the modal open so the admin can see the status change reflect in the table behind it
            }
        });

        // Start App
        initFirebase();
    </script>
